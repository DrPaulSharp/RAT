1function f = Asymconvstep(x,xw,xcen,s1,s2,h)
 2% Produces a step function convoluted with differnt error functions
 3% on each side.
 4% Convstep (x,xw,xcen,s1,s2,h)
 5%       x = vector of x values
 6%      xw = Width of step function
 7%    xcen = Centre point of step function
 8%       s1 = Roughness parameter of left side
 9%       s2 = Roughness parameter of right side
10%       h = Height of step function.
11
12% if length(xw) > 1
13%     ME = MException('VerifyOutput:OutOfBounds', ...
14%              'xw must be single value');
15%     throw(ME);
16% end
17% 
18% if length(xcen) > 1
19%     ME = MException('VerifyOutput:OutOfBounds', ...
20%              'xcen must be single value');
21%     throw(ME);
22% end
23% 
24% if length(s1) > 1
25%     ME = MException('VerifyOutput:OutOfBounds', ...
26%              's1 must be single value');
27%     throw(ME);
28% end
29% 
30% if length(s2) > 1
31%     ME = MException('VerifyOutput:OutOfBounds', ...
32%              's2 must be single value');
33%     throw(ME);
34% end
35
36r = xcen + (xw/2);
37l = xcen - (xw/2);
38
39aFactor = (2^0.5)*s1;
40bFactor = (2^0.5)*s2;
41
42a = (x-l)/aFactor;
43b = (x-r)/bFactor;
44
45f = (h/2)*(erf(a)-erf(b));
