    8   void matlabCallFun(double params[], int arrayLen, char *funName, char *pathCall, double bulkIn, double bulkOut, double contrast, double *sum, double *ans)
    9   {
   10   	static Engine *ep;
   11   	static double engStatus = 0;
   12   	mxArray *result = NULL;
   13   	mxArray *PARAMS = NULL;
   14   	mxArray *BULKIN = NULL;
   15   	mxArray *BULKOUT = NULL;
   16   	mxArray *CONTRAST = NULL;
   17   	mxArray *lays = NULL;
   18   
   19   	double* s;
   20   	double* layers;
   21   
   22   
   23   	/*Contrast = -1 means start engine only, then return*/
   24   	if(contrast == -1) {
   25   		ep = engOpen("");
   26   				if(ep==0) {
   27   					printf("Connecton to Matlab Engine failed\n");
   28   					return;
   29   				}
   30   				else {
   31   					printf("Starting Matlab Engine\n");
   32   					engStatus == 1;
   33   					return;
   34   				}
   35   	}
   36   
   37       /*Contrast = -2 means close engine only (if open), then return*/
   38   	if(contrast == -2) {
   39   				if(ep==0) {
   40   					printf("No engine open\n");
   41   				}
   42   				else {
   43   					printf("Closing Matlab Engine\n");
   44   					engClose(ep);
   45   					engStatus == 0;
   46   					return;
   47   				}
   48   	}
   49   
   50   
   51   	/*Automatically start engine if closed to catch non-initialised call*/
   52   	if(engStatus == 0) {
   53   		ep = engOpen("");		
   54   		if(ep==0) {
   55   			printf("Connecton to Matlab Engine failed\n");
   56   			return;
   57   		}
   58   		else {
   59   			printf("Starting Matlab Engine\n");
   60   			/*engEvalString(ep,(char *)pathCall);*/
   61   			/*engEvalString(ep,"cd('/home/arwel/eclipseWorkspace_new/matlabEngine_demo/src')")*/;
   62   			mxArray* MYPATH = mxCreateString(pathCall);
   63   			engPutVariable(ep, "mypath", MYPATH);
   64   			engEvalString(ep,"eval(mypath)");
   65   			mxDestroyArray(MYPATH);
   66   			engStatus = 1;
   67   		}
   68   	}
   69   
   70   	PARAMS = mxCreateDoubleMatrix(1,arrayLen,mxREAL);
   71   	memcpy(mxGetPr(PARAMS), params, arrayLen*sizeof(double));
   72   	engPutVariable(ep,"params",PARAMS);
   73   
   74   	BULKIN = mxCreateDoubleMatrix(1,1,mxREAL);
   75   	memcpy((void *)mxGetPr(BULKIN), &bulkIn, 1*sizeof(double));
   76   	engPutVariable(ep,"bulk_in",BULKIN);
   77   
   78   	BULKOUT = mxCreateDoubleMatrix(1,1,mxREAL);
   79   	memcpy((void *)mxGetPr(BULKOUT), &bulkOut, 1*sizeof(double));
   80   	engPutVariable(ep,"bulk_out",BULKOUT);
   81   
   82   	CONTRAST = mxCreateDoubleMatrix(1,1,mxREAL);
   83   	memcpy((void *)mxGetPr(CONTRAST), &contrast, 1*sizeof(double));
   84   	engPutVariable(ep,"contrast",CONTRAST);
   85   
   86   	mxArray* MYFUN = mxCreateString(funName);
   87   	engPutVariable(ep, "myfun", MYFUN);
   88   	engEvalString(ep,"eval(myfun)");
   89   	/*engEvalString(ep,"[total,layers] = debugMfile(params,bulk_in,bulk_out,contrast)");*/
   90   	mxDestroyArray(MYFUN);
   91   
   92   	result = engGetVariable(ep,"total");
   93   	s = (double *)mxGetData(result);
   94   	memcpy( sum, s, mxGetNumberOfElements(result)*sizeof(double) );
   95   
   96   	lays = engGetVariable(ep,"layers");
   97   	layers = (double *)mxGetData(lays);
   98   	memcpy(ans,layers,mxGetNumberOfElements(lays)*sizeof(double) );
   99   
  100   	/*engClose(ep)*/;
  101   
  102   	mxDestroyArray(PARAMS);
  103   	mxDestroyArray(BULKIN);
  104   	mxDestroyArray(BULKOUT);
  105   	mxDestroyArray(CONTRAST);
  106   
  107   };
