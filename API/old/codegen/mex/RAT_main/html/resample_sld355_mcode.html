1function [x_new_tot,y_new_tot] = resample_sld(sld,y_min_step)
 2
 3x = sld(:,1);
 4prof = sld(:,2);
 5
 6x_new = zeros(1e4,1);
 7y_new = zeros(1e4,1);
 8
 9thisx = x(1);
10count = 1;
11i = 2;
12while i < length(x)
