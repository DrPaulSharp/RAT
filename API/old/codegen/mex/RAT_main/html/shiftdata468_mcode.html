 3%Shifts the data according to scale factor

12switch dataPresent
13    case 1
14        if scalefac == 0
15            scalefac = 1e-30;
16        end
17        data(:,1) = data(:,1) + horshift;
18        data(:,2) = data(:,2) ./ scalefac;
19        data(:,3) = data(:,3) ./ scalefac;
20        
21        lowLimit = dataLimits(1);
22        hiLimit = dataLimits(2);
23        
24        lowIndex = find(data(:,1) < lowLimit);
25        if ~isempty(lowIndex)
26            lowIndex = lowIndex(end);
27        else
28            lowIndex = 1;
29        end
30        
31        hiIndex = find(data(:,1) > hiLimit);
32        if  ~isempty(hiIndex)
33            hiIndex = hiIndex(1);
34        else
35            hiIndex = length(data(:,1));
36        end
37        
38        shifted_data = data(lowIndex:hiIndex,:);
39    otherwise
40        shifted_data = data;
41end
