1function problem = linear_DE_new_test(problem,controls)
  2
  3F_VTR = 0.00001; %Value to reach
  4I_D = length(problem.fitpars);
  5
  6FVr_minbound = problem.fitconstr(:,1)'; 
  7FVr_maxbound = problem.fitconstr(:,2)'; 
  8I_bnd_constr = 1;  %1: use bounds as bound constraints, 0: no bound constraints
  9
 10% I_NP            number of population members
 11I_NP = 20; 
 12
 13% I_itermax       maximum number of iterations (generations)
 14I_itermax = 1000;%problem.maxiter;
 15
 16% F_weight        DE-stepsize F_weight ex [0, 2]
 17F_weight = 0.3; 
 18
 19% F_CR            crossover probabililty constant ex [0, 1]
 20F_CR = 0.9; 
 21
 22% I_strategy     1 --> DE/rand/1:
 23%                      the classical version of DE.
 24%                2 --> DE/local-to-best/1:
 25%                      a version which has been used by quite a number
 26%                      of scientists. Attempts a balance between robustness
 27%                      and fast convergence.
 28%                3 --> DE/best/1 with jitter:
 29%                      taylored for small population sizes and fast convergence.
 30%                      Dimensionality should not be too high.
 31%                4 --> DE/rand/1 with per-vector-dither:
 32%                      Classical DE with dither to become even more robust.
 33%                5 --> DE/rand/1 with per-generation-dither:
 34%                      Classical DE with dither to become even more robust.
 35%                      Choosing F_weight = 0.3 is a good start here.
 36%                6 --> DE/rand/1 either-or-algorithm:
 37%                      Alternates between differential mutation and three-point-
 38%                      recombination.           
 39
 40I_strategy = 5;
 41
 42% I_refresh     intermediate output will be produced after "I_refresh"
 43%               iterations. No intermediate output will be produced
 44%               if I_refresh is < 1
 45I_refresh = 1;
 46
 47% I_plotting    Will use plotting if set to 1. Will skip plotting otherwise.
 48I_plotting = 0;
 49
 50%-----Definition of tolerance scheme--------------------------------------
 51%-----The scheme is sampled at I_lentol points----------------------------
 52I_lentol   = 50;
 53FVr_x      = linspace(-1,1,I_lentol); %ordinate running from -1 to +1
 54FVr_lim_up = ones(1,I_lentol);   %upper limit is 1
 55FVr_lim_lo = -ones(1,I_lentol);  %lower limit is -1
 56
 57%Tell comiler abut variable sizes
 58coder.varsize('S_struct.I_lentol', [Inf 1],[1 0]);
 59coder.varsize('S_struct.FVr_x', [1 Inf],[0 1]);
 60coder.varsize('S_struct.FVr_lim_up', [1 Inf],[0 1]);
 61coder.varsize('S_struct.FVr_lim_lo', [1 Inf],[0 1]);
 62
 63coder.varsize('S_struct.I_NP', [1 1],[0 0]);
 64coder.varsize('S_struct.F_weight', [1 1],[0 0]);
 65coder.varsize('S_struct.F_CR', [1 1],[0 0]);
 66coder.varsize('S_struct.I_D', [1 1],[0 0]);
 67coder.varsize('S_struct.FVr_minbound', [1 2],[0 0]);
 68coder.varsize('S_struct.FVr_maxbound', [1 2],[0 0]);
 69coder.varsize('S_struct.I_bnd_constr', [1 1],[0 0]);
 70coder.varsize('S_struct.I_itermax', [1 1],[0 0]);
 71coder.varsize('S_struct.F_VTR', [1 1],[0 0]);
 72coder.varsize('S_struct.I_strategy', [1 1],[0 0]);
 73coder.varsize('S_struct.I_refresh', [1 1],[0 0]);
 74coder.varsize('S_struct.I_plotting', [1 1],[0 0]);
 75coder.varsize('S_struct.FM_pop',[Inf 2],[1 0]);
 76coder.varsize('FVr_bestmem',[1 2],[0 0]);
 77
 78%-----tie all important values to a structure that can be passed along----
 79S_struct.I_lentol   = I_lentol;
 80S_struct.FVr_x      = FVr_x;
 81S_struct.FVr_lim_up = FVr_lim_up;
 82S_struct.FVr_lim_lo = FVr_lim_lo;
 83
 84S_struct.I_NP         = I_NP;
 85S_struct.F_weight     = F_weight;
 86S_struct.F_CR         = F_CR;
 87S_struct.I_D          = I_D;
 88S_struct.FVr_minbound = FVr_minbound;
 89S_struct.FVr_maxbound = FVr_maxbound;
 90S_struct.I_bnd_constr = I_bnd_constr;
 91S_struct.I_itermax    = I_itermax;
 92S_struct.F_VTR        = F_VTR;
 93S_struct.I_strategy   = I_strategy;
 94S_struct.I_refresh    = I_refresh;
 95S_struct.I_plotting   = I_plotting;
 96S_struct.FM_pop = zeros(I_NP,2);
 97S_struct.FVr_bestmem = [0 0];
 98
 99res = rascal_deopt(@intrafun,problem,@PlotIt,controls,S_struct);
100
101end
