1function outLayers = allocateParamsToLayers(params, layersDetails);
 2
 3%This function takes the list of all layers (in terms of parameter
 4%numbers) in layersDetails, and loops through putting in the correct
 5%parameter value from the parameter array in each layer. 
 6
 7numberOfLayers = length(layersDetails);
 8thisOutLayer = zeros(1,5);
 9outLayers = cell(1,numberOfLayers);
10
11for i = 1:numberOfLayers
12    thisLayer = layersDetails{i};
13    for n = 1:(length(thisLayer)-1)
14        thisVal = thisLayer(n);
15        if ~isnan(thisVal)
16            thisOutLayer(n) = params(thisVal);
17        else
18            thisOutLayer(n) = NaN;
19        end
20    end
21    thisOutLayer(n+1) = thisLayer(end);
22    outLayers{i} = thisOutLayer;
23end
24
25end
