1function [outSsubs,...

  9            reflectivity,...
 10            Simulation,...
 11            shifted_data,...
 12            layerSlds,...
 13            sldProfiles,...
 14            allLayers,...
 15            allRoughs] = standardTF_stanlay_single(resample, ...
 16            numberOfContrasts, ...
 17            geometry, ...
 18            repeatLayers , ...
 19            cBacks , ...
 20            cShifts , ...
 21            cScales , ...
 22            cNbas , ...
 23            cNbss, ...
 24            cRes , ...
 25            backs , ...
 26            shifts , ...
 27            sf, ...
 28            nba , ...
 29            nbs , ...
 30            res , ...
 31            dataPresent , ...
 32            allData , ...
 33            dataLimits , ...
 34            simLimits , ...
 35            nParams , ...
 36            params , ...
 37            contrastLayers , ...
 38            numberOfLayers , ...
 39            layersDetails,...
 40            problemDef_limits);
 41
 42
 43        
 44global RAT_DEBUG DEBUGVARS        
 45        
 46%coder.extrinsic('callModules');  
 47coder.varsize('backgs',[Inf,1],[1 0]); 
 48coder.varsize('shifted_dat',[Inf 3],[1 0]);
 49
 50            
 51%Pre-Allocation...
 52backgs = zeros(numberOfContrasts,1);
 53qshifts = zeros(numberOfContrasts,1);
 54sfs = zeros(numberOfContrasts,1);
 55nbas = zeros(numberOfContrasts,1);
 56nbss = zeros(numberOfContrasts,1);
 57resols = zeros(numberOfContrasts,1);
 58allRoughs = zeros(numberOfContrasts,1);
 59outSsubs = zeros(numberOfContrasts,1);
 60chis =  zeros(numberOfContrasts,1);
 61allLayers = cell(numberOfContrasts,1); 
 62layerSlds = cell(numberOfContrasts,1);
 63sldProfiles = cell(numberOfContrasts,1);
 64shifted_data = cell(numberOfContrasts,1);
 65reflectivity = cell(numberOfContrasts,1);
 66Simulation = cell(numberOfContrasts,1);
 67boxes = cell(1,1);
 68
 69%coder preallocation - allLayers may be legacy and not used..
 70for n = 1:length(allLayers)
 71    allLayers{n} = 0;
 72end
 73
 74
 75
 76outParameterisedLayers = allocateParamsToLayers(params, layersDetails);
 77
 78for i = 1:numberOfContrasts
 79    [backgs(i),qshifts(i),sfs(i),nbas(i),nbss(i),resols(i)] = backSort(cBacks(i),cShifts(i),cScales(i),cNbas(i),cNbss(i),cRes(i),backs,shifts,sf,nba,nbs,res);
 80
 81    allRoughs(i) = params(1);
 82    thisContrastLayers = allocateLayersForContrast(contrastLayers{i},outParameterisedLayers);
 83    [outLayers, outSsubs(i)] = groupLayers_Mod(thisContrastLayers,allRoughs(i),geometry,nbas(i),nbss(i));
 84    
 85    sldProfile = makeSLDProfiles(nbas(i),nbss(i),outLayers,outSsubs(i),repeatLayers{i});
 86    sldProfiles{i} = sldProfile;
 87
 88    if resample(i) == 1
 89        layerSld = resampleLayers(sldProfile);
 90        layerSlds{i} = layerSld;
 91    else
 92        layerSld = outLayers;
 93        layerSlds{i} = layerSld;
 94    end
 95   
 96    shifted_dat = shiftdata(sfs(i),qshifts(i),dataPresent(i),allData{i},dataLimits{i});
 97    shifted_data{i} = shifted_dat;
 98    
 99    [reflect,Simul] = callReflectivity(nbas(i),nbss(i),simLimits{i},repeatLayers{i},shifted_dat,layerSld,outSsubs(i),backgs(i),resols(i));
100    reflectivity{i} = reflect;
101    Simulation{i} = Simul;
102    
103    
104    chis(i) = chiSquared(shifted_dat,reflect,nParams);
105end
