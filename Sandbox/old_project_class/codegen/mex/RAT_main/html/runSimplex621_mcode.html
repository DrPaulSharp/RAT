1function [problem,result] = runSimplex(problemDef,problemDef_cells,problemDef_limits,controls)
  2
  3numberOfContrasts = problemDef.numberOfContrasts;
  4preAlloc = zeros(numberOfContrasts,1);
  5
  6result = cell(6,1);
  7
  8problem = struct('ssubs',preAlloc,...
  9                 'backgrounds',preAlloc,...
 10                 'qshifts',preAlloc,...
 11                 'scalefactors',preAlloc,...
 12                 'nbairs',preAlloc,...
 13                 'nbsubs',preAlloc,...
 14                 'resolutions',preAlloc,...
 15                 'calculations',struct('all_chis',preAlloc,'sum_chi',0),...
 16                 'allSubRough',preAlloc);
 17
 18
 19[problemDef,fitNames] = fitsetup(problemDef,problemDef_cells,problemDef_limits,controls);
 20
 21maxIter = controls.algArgs.maxInter;
 22tolFun = controls.algArgs.tolFun;
 23tolX = controls.algArgs.tolX;
 24maxFunEvals = controls.algArgs.maxFunEvals;
 25
 26options = optimset('MaxIter',maxIter,'TolFun',tolFun,'TolX',tolX,'MaxFunEvals',maxFunEvals,'Display','iter');
 27
 28x0 = problemDef.fitpars;
 29LB = problemDef.fitconstr(:,1);
 30UB = problemDef.fitconstr(:,2);
 31
 32% size checks
 33xsize = size(x0);
 34x0 = x0(:);
 35n=length(x0);
 36
 37if isempty(LB)
 38  LB = repmat(-inf,n,1);
 39else
 40  LB = LB(:);
 41end
 42if isempty(UB)
 43  UB = repmat(inf,n,1);
 44else
 45  UB = UB(:);
 46end
 47
 48
 49
 50% stuff into a struct to pass around
 51params.args = [];%varargin;
 52params.LB = LB;
 53params.UB = UB;
 54params.fun = problemDef.module;%fun;
 55
 56% 0 --> unconstrained variable
 57% 1 --> lower bound only
 58% 2 --> upper bound only
 59% 3 --> dual finite bounds
 60params.BoundClass = zeros(n,1);
 61for i=1:n
 62  k = isfinite(LB(i)) + 2*isfinite(UB(i));
 63  params.BoundClass(i) = k;
 64  if (k==3) && (LB(i)==UB(i))
 65    error 'Bounds must be distinct if both are supplied.'
 66  end
 67end
 68
 69% transform starting values into their unconstrained
 70% surrogates. Check for infeasible starting guesses.
 71x0u = x0;
 72for i = 1:n
 73  switch params.BoundClass(i)
 74    case 1
 75      % lower bound only
 76      if x0(i)<=LB(i)
 77        % infeasible starting value. Use bound.
 78        x0u(i) = 0;
 79      else
 80        x0u(i) = sqrt(x0(i) - LB(i));
 81      end
 82    case 2
 83      % upper bound only
 84      if x0(i)>=UB(i)
 85        % infeasible starting value. use bound.
 86        x0u(i) = 0;
 87      else
 88        x0u(i) = sqrt(UB(i) - x0(i));
 89      end
 90    case 3
 91      % lower and upper bounds
 92      if x0(i)<=LB(i)
 93        % infeasible starting value
 94        x0u(i) = -pi/2;
 95      elseif x0(i)>=UB(i)
 96        % infeasible starting value
 97        x0u(i) = pi/2;
 98      else
 99        x0u(i) = 2*(x0(i) - LB(i))/(UB(i)-LB(i)) - 1;
100        x0u(i) = asin(max(-1,min(1,x0u(i))));
101      end
102    case 0
103      % unconstrained variable. x0u(i) is set.
104  end
105end
106
107% now we can call fminsearch, but with our own
108% intra-objective function.
109
110[xu,fval,exitflag,output] = fminsearch(@simplexIntrafun,x0u,options,problemDef,problemDef_cells,problemDef_limits,controls,params,300);
111
112%[xu,fval,exitflag,output] = simplex(@simplexIntrafun,x0u,problemDef,problemDef_cells,problemDef_limits,controls,options,params,300);
113
114% undo the variable transformations into the original space
115x = simplex_xtransform(xu,params);
116
117% final reshape
118%x = reshape(x,xsize);
119
120problemDef.fitpars = x;
121problemDef = unpackparams(problemDef,controls);
122[problem,result] = reflectivity_calculation(problemDef,problemDef_cells,problemDef_limits,controls);
123
124end
