1function [x_new_tot,y_new_tot] = resample_sld(sld,y_min_step);
 2
 3%coder.varsize('sld',[1e4 2],[1 0]);
 4
 5x = sld(:,1);
 6prof = sld(:,2);
 7
 8x_new = zeros(1e4,1);
 9y_new = zeros(1e4,1);
10
11thisx = x(1);
12count = 1;
13i = 2;
14while i < length(x);
15    this_step = prof(i) - prof(i-1);
16    if abs(this_step) < y_min_step;
17        %Keep original points
18        x_new(count) = x(i);
19        y_new(count) = prof(i);
20        count = count + 1;
21        i = i + 1;
22    else
23        if this_step > 0
24            newsteps = prof(i-1):y_min_step:prof(i);
25        else
26            newsteps = prof(i-1):-y_min_step:prof(i);
27        end
28        new_xs = interp1(prof(i-1:i),x(i-1:i),newsteps);
29        x_new = [x_new new_xs];
30        y_new = [y_new newsteps];
31        count = length(x_new);
32        i = i + 1;
33    end
34end
35%x_new_tot = x_new(:);
36%y_new_tot = y_new(:);
37
38x_new = x_new(1:count,1);
39y_new = y_new(1:count,1);
40
41%Make it into a 'histogram'
42x_new_tot = zeros(count,1);
43y_new_tot = zeros(count,1);
44for i =1:length(x_new)-1;
45    x_new_tot(i) = x_new(i + 1) - x_new(i);
46    y_new_tot(i) = y_new(i);
47end
48
49
50%Remove any zero thickness layers...
51good_layers = find(x_new_tot ~= 0);
52x_new_tot = x_new_tot(good_layers);
53y_new_tot = y_new_tot(good_layers);
54
